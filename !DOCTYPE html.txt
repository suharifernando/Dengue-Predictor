  <script>
    function getVal(id) {
      return document.getElementById(id).value.trim();
    }

    function getNum(id) {
      const v = getVal(id);
      return v === '' ? NaN : parseFloat(v);
    }

    function getInt(id) {
      const v = getVal(id);
      return v === '' ? 0 : parseInt(v, 10);
    }

    function showError(msg) {
      const el = document.getElementById('error-msg');
      el.textContent = msg;
      el.style.display = 'block';
    }

    function clearError() {
      const el = document.getElementById('error-msg');
      el.textContent = '';
      el.style.display = 'none';
    }

    function calcRisk(data) {
      let score = 0;
      const t = data.temperature;
      const r = data.rainfall;
      const h = data.humidity;

      // Temperature scoring
      if (t >= 25 && t <= 32) {
        score += 3;
      } else if ((t >= 23 && t < 25) || (t > 32 && t <= 34)) {
        score += 2;
      } else {
        score += 1;
      }

      // Rainfall scoring
      if (r >= 80 && r <= 200) {
        score += 3;
      } else if ((r >= 30 && r < 80) || (r > 200 && r <= 300)) {
        score += 2;
      } else {
        score += 1;
      }

      // Humidity scoring
      if (h >= 70 && h <= 90) {
        score += 3;
      } else if ((h >= 60 && h < 70) || (h > 90 && h <= 95)) {
        score += 2;
      } else {
        score += 1;
      }

      // Standing water
      if (data.standingWater >= 3) score += 2;
      else if (data.standingWater >= 1) score += 1;

      // Recent cases
      if (data.recentCases >= 20) score += 3;
      else if (data.recentCases >= 5) score += 2;
      else if (data.recentCases > 0) score += 1;

      // Urbanization
      if (data.urbanization === 'urban') score += 1;

      // Max possible score = 13
      const prob = Math.min(0.98, Math.max(0.02, score / 13));

      let level, levelClass, badgeClass, barClass, tips;

      if (prob >= 0.75) {
        level = 'Very High Risk';
        levelClass = 'level-vhigh';
        badgeClass = 'badge-vhigh';
        barClass = 'bar-vhigh';
        tips = [
          'Eliminate all standing water immediately',
          'Use mosquito repellent at all times',
          'Visit a clinic if fever, rash, or joint pain develops',
          'Report cases to the local health authority',
          'Avoid outdoor activity at dawn and dusk'
        ];
      } else if (prob >= 0.55) {
        level = 'High Risk';
        levelClass = 'level-high';
        badgeClass = 'badge-high';
        barClass = 'bar-high';
        tips = [
          'Check and drain all stagnant water containers',
          'Use bed nets and fit window screens',
          'Wear long sleeves during peak mosquito hours',
          'Monitor for fever and flu-like symptoms'
        ];
      } else if (prob >= 0.35) {
        level = 'Moderate Risk';
        levelClass = 'level-mod';
        badgeClass = 'badge-mod';
        barClass = 'bar-mod';
        tips = [
          'Monitor local health bulletins regularly',
          'Keep gutters and drains clear of blockages',
          'Use insect repellent when outdoors',
          'Remove unused containers that collect water'
        ];
      } else {
        level = 'Low Risk';
        levelClass = 'level-low';
        badgeClass = 'badge-low';
        barClass = 'bar-low';
        tips = [
          'Maintain routine mosquito prevention measures',
          'Keep surroundings clean and dry',
          'Stay informed about dengue trends in your area'
        ];
      }

      return { prob, level, levelClass, badgeClass, barClass, tips };
    }

    function predict() {
      clearError();

      const district      = getVal('district');
      const temperature   = getNum('temperature');
      const rainfall      = getNum('rainfall');
      const humidity      = getNum('humidity');
      const recentCases   = getInt('recentCases');
      const standingWater = getInt('standingWater');
      const urbanization  = getVal('urbanization');

      // Validation
      if (!district) {
        showError('Please select a district.');
        return;
      }
      if (isNaN(temperature)) {
        showError('Please enter the temperature.');
        return;
      }
      if (temperature < 0 || temperature > 60) {
        showError('Temperature must be between 0 and 60 °C.');
        return;
      }
      if (isNaN(rainfall)) {
        showError('Please enter the rainfall amount.');
        return;
      }
      if (rainfall < 0) {
        showError('Rainfall cannot be negative.');
        return;
      }
      if (isNaN(humidity)) {
        showError('Please enter the humidity.');
        return;
      }
      if (humidity < 0 || humidity > 100) {
        showError('Humidity must be between 0 and 100%.');
        return;
      }

      const result = calcRisk({ temperature, rainfall, humidity, recentCases, standingWater, urbanization });
      const pct = Math.round(result.prob * 100);

      const tipItems = result.tips.map(t => `<li>${t}</li>`).join('');

      document.getElementById('result-panel').innerHTML = `
        <div class="risk-header">
          <span class="risk-pct ${result.levelClass}">${pct}%</span>
          <span class="risk-badge ${result.badgeClass}">${result.level}</span>
        </div>
        <div class="risk-bar-bg">
          <div class="risk-bar-fill ${result.barClass}" style="width: ${pct}%"></div>
        </div>
        <p class="risk-description">
          Assessment for <strong>${district}</strong> based on the entered environmental
          and epidemiological conditions.
        </p>
        <div class="tips-section">
          <p class="tips-title">Recommended Actions</p>
          <ul class="tips-list">${tipItems}</ul>
        </div>
      `;
    }
  </script>
